### Hour 3-4: ProfileMessageBus Implementation

```typescript
// packages/runtime/src/profile-bus.ts
export type DomainEvent =
  | { type: 'chat-state-replaced'; publicKeyHex: string; profileId: string }
  | { type: 'group-invite-accepted'; groupId: string; memberPubkey: string; relayUrl?: string }
  | { type: 'group-invite-received'; invite: CommunityInvite }
  | { type: 'new-message'; conversationId: string; message: Message }
  | { type: 'membership-updated'; groupId: string; members: PublicKeyHex[] }
  | { type: 'relay-connected'; relayUrl: string }
  | { type: 'relay-disconnected'; relayUrl: string; reason: string };

export type EventHandler = (event: DomainEvent) => void;
export type Unsubscribe = () => void;

export interface ProfileMessageBus {
  publish(event: DomainEvent): void;
  subscribe(handler: EventHandler): Unsubscribe;
  subscribeTo<T extends DomainEvent['type']>(
    type: T,
    handler: (event: Extract<DomainEvent, { type: T }>) => void
  ): Unsubscribe;
}

export function createProfileMessageBus(params: {
  profileId: string;
}): ProfileMessageBus {
  const handlers = new Set<EventHandler>();
  
  return {
    publish: (event) => {
      handlers.forEach(handler => {
        try {
          handler(event);
        } catch (e) {
          console.error(`[ProfileBus ${params.profileId}] Handler error:`, e);
        }
      });
    },
    
    subscribe: (handler) => {
      handlers.add(handler);
      return () => handlers.delete(handler);
    },
    
    subscribeTo: (type, handler) => {
      const wrapper = (event: DomainEvent) => {
        if (event.type === type) {
          handler(event as any);
        }
      };
      handlers.add(wrapper);
      return () => handlers.delete(wrapper);
    }
  };
}
```

import { createProfileMessageBus, ProfileMessageBus } from '../profile-bus';

const ProfileContext = createContext<string | null>(null);
const BusContext = createContext<ProfileMessageBus | null>(null);

export function ProfileProvider({ 
  profileId, 
  children 
}: { 
  profileId: string; 
  children: React.ReactNode;
}) {
  const bus = useMemo(() => createProfileMessageBus({ profileId }), [profileId]);
  
  return (
    <ProfileContext.Provider value={profileId}>
      <BusContext.Provider value={bus}>
        {children}
      </BusContext.Provider>
    </ProfileContext.Provider>
  );
}

export function useProfileId(): string {
  const profileId = useContext(ProfileContext);
  if (!profileId) throw new Error('useProfileId must be used within ProfileProvider');
  return profileId;
}

export function useProfileBus(): ProfileMessageBus {
  const bus = useContext(BusContext);
  if (!bus) throw new Error('useProfileBus must be used within ProfileProvider');
  return bus;
}

import { describe, it, expect, vi } from 'vitest';
import { createProfileMessageBus } from './profile-bus';

describe('ProfileMessageBus', () => {
  it('delivers events to subscribers', () => {
    const bus = createProfileMessageBus({ profileId: 'test' });
    const handler = vi.fn();
    
    bus.subscribe(handler);
    bus.publish({ type: 'relay-connected', relayUrl: 'wss://test.relay' });

      relayUrl: 'wss://test.relay'
    });

  it('isolates events by profile', () => {
    const busA = createProfileMessageBus({ profileId: 'A' });
    const busB = createProfileMessageBus({ profileId: 'B' });
    const handlerB = vi.fn();
    
    busB.subscribe(handlerB);
    busA.publish({ type: 'relay-connected', relayUrl: 'wss://test.relay' });
    
    expect(handlerB).not.toHaveBeenCalled();
  });
  
  it('supports typed subscriptions', () => {
    const bus = createProfileMessageBus({ profileId: 'test' });
    const handler = vi.fn();
    
    bus.subscribeTo('new-message', handler);
    bus.publish({ type: 'relay-connected', relayUrl: 'wss://test.relay' });
    bus.publish({ type: 'new-message', conversationId: 'test', message: {} as any });
    
    expect(handler).toHaveBeenCalledTimes(1);
  });
  
  it('allows unsubscribe', () => {
    const bus = createProfileMessageBus({ profileId: 'test' });
    const handler = vi.fn();
    
    const unsubscribe = bus.subscribe(handler);
    unsubscribe();
    bus.publish({ type: 'relay-connected', relayUrl: 'wss://test.relay' });
    
    expect(handler).not.toHaveBeenCalled();
  });
});

// apps/pwa/app/test/single-process-a-b.test.ts
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { createAppContainer } from '../container';

describe('Single Process A/B Isolation', () => {
  let containerA: ReturnType<typeof createAppContainer>;
  let containerB: ReturnType<typeof createAppContainer>;

  beforeEach(() => {
    containerA = createAppContainer({ profileId: 'profile-a' });
    containerB = createAppContainer({ profileId: 'profile-b' });
  });

  afterEach(() => {
    // Cleanup
    containerA.destroy?.();
    containerB.destroy?.();
  });

  it('isolates chat state between profiles', () => {
    const publicKey = 'test-pubkey';
    
    // A saves state
    containerA.stores.chat.save(publicKey, { messages: [{ id: 'msg-a', content: 'Hello from A' }] });
    
    // B loads state - should not see A's messages
    const bState = containerB.stores.chat.load(publicKey);
    
    expect(bState?.messages).toHaveLength(0);
  });

  it('isolates bus events between profiles', () => {
    const eventsA: any[] = [];
    const eventsB: any[] = [];
    
    containerA.bus.subscribe((e) => eventsA.push(e));
    containerB.bus.subscribe((e) => eventsB.push(e));
    
    // A publishes event
    containerA.bus.publish({ type: 'relay-connected', relayUrl: 'wss://test.relay' });
    
    // Only A should receive
    expect(eventsA).toHaveLength(1);
    expect(eventsB).toHaveLength(0);
  });

  it('isolates global state replacements', () => {
    const publicKey = 'test-pubkey';
    let aReceived = false;
    let bReceived = false;
    
    containerA.bus.subscribeTo('chat-state-replaced', () => { aReceived = true; });
    containerB.bus.subscribeTo('chat-state-replaced', () => { bReceived = true; });
    
    containerA.stores.chat.save(publicKey, { messages: [] });
    
    expect(aReceived).toBe(true);
    expect(bReceived).toBe(false);
  });
});
